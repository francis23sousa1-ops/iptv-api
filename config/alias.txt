# Lista de aliases (apelidos) para nomes de canais.
# Formato: NomePrincipal,Alias1,Alias2
# Suporta expressões regulares usando o prefixo re: para capturar dezenas de variações automaticamente.

; =========================
; Canais Abertos
; =========================
Globo,re:(?i)^\s*(Rede\s*|TV\s*)?Globo[\s\S]*$
SBT,re:(?i)^\s*(Canal\s*|TV\s*)?SBT[\s\S]*$,Sistema Brasileiro de Televisao
Record,re:(?i)^\s*(Rede\s*|TV\s*)?Record[\s\S]*$,RecordTV
Band,re:(?i)^\s*(Rede\s*|TV\s*)?Band(eirantes)?[\s\S]*$
RedeTV!,re:(?i)^\s*Rede\s*TV[\s\S]*$

; =========================
; Canais de Esportes
; =========================
# O código (?![0-9]) garante que a busca pela ESPN 1 não puxe acidentalmente a ESPN 2 ou 3.
SporTV,re:(?i)^\s*SporTV[-\s_]*0?1?(?![0-9])[\s\S]*$
SporTV 2,re:(?i)^\s*SporTV[-\s_]*0?2[\s\S]*$
SporTV 3,re:(?i)^\s*SporTV[-\s_]*0?3[\s\S]*$

ESPN,re:(?i)^\s*ESPN[-\s_]*0?1?(?![0-9])[\s\S]*$,ESPN Brasil,ESPN BR
ESPN 2,re:(?i)^\s*ESPN[-\s_]*0?2[\s\S]*$
ESPN 3,re:(?i)^\s*ESPN[-\s_]*0?3[\s\S]*$
ESPN 4,re:(?i)^\s*ESPN[-\s_]*0?4[\s\S]*$
BandSports,re:(?i)^\s*Band[-\s_]*Sports[\s\S]*$
Premiere,re:(?i)^\s*Premiere(?![0-9])[\s\S]*$,Premiere Clubes

HBO 2,re:(?i)^\s*HBO[-\s_]*0?2[\s\S]*$
HBO Family,re:(?i)^\s*HBO[-\s_]*Fam[\s\S]*$

Warner Channel,re:(?i)^\s*Warner[\s\S]*$,WB Channel
Star Channel,re:(?i)^\s*Star[-\s_]*(Channel|TV)?[\s\S]*$,Fox Channel,Fox

Space,re:(?i)^\s*Space[\s\S]*$
TNT,re:(?i)^\s*TNT[\s\S]*$,TNT Brasil
